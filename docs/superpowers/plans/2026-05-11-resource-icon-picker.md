# Resource Icon Picker Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a resource icon picker that searches the local Iconify library from the server, supports configured Chinese search aliases, and lets users select or clear icons from the resource form without hand-editing icon names.

**Architecture:** Shared zod schemas define the icon search query and response contract. The server builds a cached in-process metadata index from local `@iconify/json`, uses `fzf` for fuzzy recall, then applies admin-oriented ranking. The client adds a focused `ResourceIconPicker.vue` component that opens a controlled `NPopover` from the display input, renders icon-only results with tooltip names, and binds back into `ResourceFormDrawer.vue`.

**Tech Stack:** Vue 3 Composition API, Naive UI, `@iconify/vue`, VueUse, Hono, zod, `@iconify/json`, `@iconify/utils`, `fzf`, Vitest, pnpm workspace.

const iconSearchLimitMax = 100
const iconNamePattern = /^[a-z0-9]+(?:-[a-z0-9]+)*:[a-z0-9]+(?:-[a-z0-9]+)*$/

const iconSearchLimitSchema = z
  .string()
  .optional()
  .default(String(iconSearchLimitDefault))
  .transform((value) => Number(value))
  .pipe(z.number().int().min(1))
  .transform((value) => Math.min(value, iconSearchLimitMax))

export const iconSearchQuerySchema = z.object({
  keyword: z
    .string()
    .optional()
    .default('')
    .transform((value) => value.trim()),
  limit: iconSearchLimitSchema,
})

export const iconSearchItemSchema = z.object({
  icon: z.string().regex(iconNamePattern, '图标名称无效'),
  prefix: z.string().min(1),
  name: z.string().min(1),
  collection: z.string().min(1),
  palette: z.boolean(),
})

export const iconSearchResponseSchema = z.object({
  list: iconSearchItemSchema.array(),
})

export type IconSearchQuery = z.infer<typeof iconSearchQuerySchema>
export type IconSearchItem = z.infer<typeof iconSearchItemSchema>
export type IconSearchResponse = z.infer<typeof iconSearchResponseSchema>

- [ ] **Step 4: Create search service**

Create `apps/server/src/modules/icons/search-service.ts`:

```ts
import collections from '@iconify/json/collections.json' with { type: 'json' }
import { lookupCollection } from '@iconify/json'
import type { IconifyJSON } from '@iconify/types'
import type { IconSearchItem, IconSearchQuery, IconSearchResponse } from '@rev30/shared'
import { AsyncFzf, asyncExtendedMatch } from 'fzf'
import {
  chineseIconSearchAliases,
  deprioritizedIconPrefixes,
  iconSearchAliasGroups,
  preferredIconPrefixes,
  recommendedIconNames,
} from './search-config'

type CollectionInfo = {
  name: string
  category?: string
  palette?: boolean
}

type IconSearchIndexItem = IconSearchItem & {
  category: string | null
  searchText: string
}

type IconSearchIndex = {
  items: IconSearchIndexItem[]
  byIcon: Map<string, IconSearchIndexItem>
  fastFinder: AsyncFzf<IconSearchIndexItem>
  extendedFinder: AsyncFzf<IconSearchIndexItem>
}

const maxCandidateCount = 12
let searchIndexPromise: Promise<IconSearchIndex> | null = null

function createItem(prefix: string, name: string, collection: CollectionInfo): IconSearchIndexItem {
  const icon = `${prefix}:${name}`
  const category = collection.category ?? null

  return {
    icon,
    prefix,
    name,
    collection: collection.name,
    category,
    palette: collection.palette ?? false,
    searchText: `${icon} ${name} ${prefix} ${collection.name} ${category ?? ''}`.toLowerCase(),
  }
}

async function buildSearchIndex(): Promise<IconSearchIndex> {
  const items: IconSearchIndexItem[] = []

  await Promise.all(
    Object.entries(collections as Record<string, CollectionInfo>).map(async ([prefix, collection]) => {
      const iconSet = (await lookupCollection(prefix)) as IconifyJSON | null

      if (!iconSet) {
        return
      }

      for (const name of Object.keys(iconSet.icons)) {
        items.push(createItem(prefix, name, collection))
      }
    }),
  )

  const byIcon = new Map(items.map((item) => [item.icon, item]))

  return {
    items,
    byIcon,
    fastFinder: new AsyncFzf(items, {
      casing: 'case-insensitive',
      fuzzy: 'v1',
      selector: (item) => item.searchText,
    }),
    extendedFinder: new AsyncFzf(items, {
      casing: 'case-insensitive',
      match: asyncExtendedMatch,
      selector: (item) => item.searchText,
    }),
  }
}

async function getSearchIndex() {
  searchIndexPromise ??= buildSearchIndex()

  return await searchIndexPromise
}

function tokenize(keyword: string) {
  return keyword.toLowerCase().split(/\s+/).filter(Boolean)
}

function addSingularPluralForms(candidates: Set<string>, token: string) {
  candidates.add(token)

  if (token.endsWith('s') && token.length > 1) {
    candidates.add(token.slice(0, -1))
    return
  }

  candidates.add(`${token}s`)
}

function addEnglishAliases(candidates: Set<string>, token: string) {
  addSingularPluralForms(candidates, token)

  const group = iconSearchAliasGroups.find((aliases) => aliases.includes(token))

  if (!group) {
    return
  }

  for (const alias of group) {
    addSingularPluralForms(candidates, alias)
  }
}

function expandKeyword(keyword: string) {
  const normalizedKeyword = keyword.trim().toLowerCase()
  const candidates = new Set<string>()

  if (!normalizedKeyword) {
    return []
  }

  candidates.add(normalizedKeyword)

  for (const token of tokenize(normalizedKeyword)) {
    addEnglishAliases(candidates, token)
  }

  for (const [chineseKeyword, aliases] of Object.entries(chineseIconSearchAliases)) {
    if (!keyword.includes(chineseKeyword)) {
      continue
    }

    for (const alias of aliases) {
      addEnglishAliases(candidates, alias)
    }
  }

  return [...candidates].filter(Boolean).slice(0, maxCandidateCount)
}

function getNameTokens(name: string) {
  return name.split('-').filter(Boolean)
}

function scoreItem(item: IconSearchIndexItem, keyword: string, candidates: string[]) {
  const normalizedKeyword = keyword.toLowerCase()
  const tokens = getNameTokens(item.name)
  let score = 0

  if (item.icon === normalizedKeyword) {
    score += 1000
  }

  if (item.name === normalizedKeyword) {
    score += 800
  }

  if (tokens.some((token) => token === normalizedKeyword || token.startsWith(normalizedKeyword))) {
    score += 360
  }

  if (item.name.includes(normalizedKeyword)) {
    score += 220
  }

  if (candidates.some((candidate) => item.name.includes(candidate))) {
    score += 160
  }

  score += preferredIconPrefixes.get(item.prefix) ?? 0

  if (item.palette) {
    score -= 30
  }

  if (deprioritizedIconPrefixes.has(item.prefix)) {
    score -= 80
  }

  return score
}

function toResponseItem(item: IconSearchIndexItem): IconSearchItem {
  return {
    icon: item.icon,
    prefix: item.prefix,
    name: item.name,
    collection: item.collection,
    palette: item.palette,
  }
}

function getRecommendedIcons(index: IconSearchIndex, limit: number): IconSearchResponse {
  return {
    list: recommendedIconNames
      .map((icon) => index.byIcon.get(icon))
      .filter((item): item is IconSearchIndexItem => item !== undefined)
      .slice(0, limit)
      .map(toResponseItem),
  }
}

export async function searchIcons(query: IconSearchQuery): Promise<IconSearchResponse> {
  const index = await getSearchIndex()
  const keyword = query.keyword.trim()

  if (!keyword) {
    return getRecommendedIcons(index, query.limit)
  }

  const candidates = expandKeyword(keyword)
  const finder = candidates.length > 1 ? index.extendedFinder : index.fastFinder
  const searchQuery = candidates.length > 1 ? candidates.join(' | ') : candidates[0]!
  const result = await finder.find(searchQuery)
  const fzfOrder = new Map(result.map((entry, order) => [entry.item.icon, order]))

  return {
    list: result
      .map((entry) => entry.item)
      .sort((left, right) => {
        const scoreDiff = scoreItem(right, keyword, candidates) - scoreItem(left, keyword, candidates)

        if (scoreDiff !== 0) {
          return scoreDiff
        }

        return (fzfOrder.get(left.icon) ?? 0) - (fzfOrder.get(right.icon) ?? 0)
      })
      .slice(0, query.limit)
      .map(toResponseItem),
  }
}
```

<script setup lang="ts">
import { Icon } from '@iconify/vue'
import { NAlert, NEmpty, NInput, NInputGroup, NPopover, NSpin, NTooltip } from 'naive-ui'
import { nextTick, ref, watch } from 'vue'
import { useDebounceFn } from '@vueuse/core'
import type { IconSearchItem } from '@rev30/shared'
import { searchIcons } from '.'

const props = defineProps<{
  value: string | null
}>()

const emit = defineEmits<{
  'update:value': [value: string | null]
  blur: []
}>()

const limit = 60
const showPanel = ref(false)
const keyword = ref('')
const isSearching = ref(false)
const searchError = ref<string | null>(null)
const results = ref<IconSearchItem[]>([])
const searchInputRef = ref<InstanceType<typeof NInput> | null>(null)

function blockDisplayInput() {
  return false
}

function handleDisplayValueUpdate(value: string) {
  if (value === '') {
    emit('update:value', null)
  }
}

async function runSearch() {
  isSearching.value = true
  searchError.value = null

  try {
    const response = await searchIcons({
      keyword: keyword.value,
      limit,
    })

    results.value = response.list
  } catch {
    searchError.value = '图标搜索失败'
  } finally {
    isSearching.value = false
  }
}

const debouncedSearch = useDebounceFn(runSearch, 200)

async function openPanel() {
  if (showPanel.value) {
    return
  }

  showPanel.value = true
  await runSearch()
  await nextTick()
  searchInputRef.value?.focus()
}

function closePanel() {
  showPanel.value = false
  emit('blur')
}

function selectIcon(icon: string) {
  emit('update:value', icon)
  closePanel()
}

watch(keyword, () => {
  if (!showPanel.value) {
    return
  }

  void debouncedSearch()
})
</script>

<template>
  <NPopover
    :show="showPanel"
    trigger="manual"
    placement="bottom-start"
    :show-arrow="false"
    @clickoutside="closePanel"
  >
    <template #trigger>
      <NInputGroup>
        <span
          class="flex w-14 shrink-0 items-center justify-center border border-r-0 border-stone-200 bg-stone-50 text-stone-500 dark:border-zinc-700 dark:bg-zinc-800 dark:text-zinc-300"
        >
          <Icon v-if="value" :icon="value" />
          <span v-else data-test="resource-icon-empty">无</span>
        </span>
        <NInput
          data-test="resource-form-icon"
          clearable
          :allow-input="blockDisplayInput"
          :value="value ?? ''"
          placeholder="未选择图标"
          @click="openPanel"
          @focus="openPanel"
          @update:value="handleDisplayValueUpdate"
        />
      </NInputGroup>
    </template>

    <div class="w-86 max-w-[calc(100vw-48px)]">
      <NInput
        ref="searchInputRef"
        v-model:value="keyword"
        data-test="resource-icon-search"
        clearable
        placeholder="搜索图标"
      />

      <NAlert v-if="searchError" class="mt-3" type="error" :show-icon="false">
        {{ searchError }}
      </NAlert>

      <NSpin :show="isSearching">
        <div v-if="results.length" class="mt-3 grid grid-cols-8 gap-2">
          <NTooltip v-for="item in results" :key="item.icon" trigger="hover">
            <template #trigger>
              <button
                type="button"
                data-test="resource-icon-option"
                class="grid size-9 place-items-center rounded border border-transparent text-lg text-stone-600 transition-colors hover:border-primary hover:bg-primary/10 hover:text-primary focus-visible:border-primary focus-visible:bg-primary/10 focus-visible:text-primary focus-visible:outline-none dark:text-zinc-300"
                :class="item.icon === value ? 'border-primary bg-primary/10 text-primary' : ''"
                :aria-label="item.icon"
                @click="selectIcon(item.icon)"
              >
                <Icon :icon="item.icon" />
              </button>
            </template>
            {{ item.icon }}
          </NTooltip>
        </div>

        <NEmpty v-else-if="!isSearching" class="py-6" description="未找到图标" />
      </NSpin>
    </div>
  </NPopover>
</template>
