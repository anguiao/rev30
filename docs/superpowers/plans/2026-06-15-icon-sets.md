# Icon Sets Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the content-management icon library with read-only built-in icon browsing and managed custom icon sets that are immediately available to runtime Iconify loading and icon search.

**Architecture:** Keep runtime icon data under `/api/icons/*` and add management/browsing APIs under `/api/icon-sets/*`, with server code organized inside `apps/server/src/modules/icons/sets`. Built-in icons remain read-only from `@iconify/json`; custom icon sets are persisted in `custom_icon_sets` and `custom_icon_set_icons`, parsed from uploaded SVG into Iconify-compatible rows. The client page renders SVG bodies directly from list APIs in fixed-size grid cells, while existing Iconify runtime and resource picker flows keep their public contracts.

**Tech Stack:** Vue 3, Naive UI, Pinia Colada, Hono, Drizzle, PGlite/PostgreSQL, zod, `@iconify/json`, `@iconify/tools`, `@iconify/types`, Vitest, pnpm workspace.

const iconSetPageSizeDefault = 20
const iconSetIconPageSizeDefault = 80
const iconSetPageSizeMax = 100
const iconSetKeywordMaxLength = 120
const customIconSetNameMaxLength = 80
const customIconSetDescriptionMaxLength = 300

const iconSetPageQuerySchema = z
  .string()
  .optional()
  .default('1')
  .transform((value) => Number(value))
  .pipe(z.number().int().min(1))

function pageSizeQuerySchema(defaultPageSize: number) {
  return z
    .string()
    .optional()
    .default(String(defaultPageSize))
    .transform((value) => Number(value))
    .pipe(z.number().int().min(1))
    .transform((value) => Math.min(value, iconSetPageSizeMax))
}

const iconSetKeywordQuerySchema = z
  .string()
  .optional()
  .transform((value) => {
    const keyword = value?.trim()
    return keyword ? keyword : undefined
  })
  .pipe(z.string().max(iconSetKeywordMaxLength, '搜索关键词过长').optional())

const iconSetPrefixSchema = z
  .string()
  .regex(new RegExp(`^${iconifyIconNamePartPatternSource}$`), '图标集前缀无效')

const iconSetIconNameSchema = z
  .string()
  .regex(new RegExp(`^${iconifyIconNamePartPatternSource}$`), '图标名称无效')

export const iconSetPrefixParamSchema = z.object({
  prefix: iconSetPrefixSchema,
})

export const iconSetSvgIconNameParamSchema = z.object({
  name: iconSetIconNameSchema,
})

export const iconSetListQuerySchema = z.object({
  keyword: iconSetKeywordQuerySchema,
  page: iconSetPageQuerySchema,
  pageSize: pageSizeQuerySchema(iconSetPageSizeDefault),
})

export const iconSetIconListQuerySchema = iconSetListQuerySchema.extend({
  prefix: iconSetPrefixSchema.optional(),
  pageSize: pageSizeQuerySchema(iconSetIconPageSizeDefault),
})

export const builtinIconSetItemSchema = z.object({
  prefix: iconSetPrefixSchema,
  name: z.string().min(1),
  total: z.number().int().min(0),
})

export const builtinIconSetListResponseSchema = z.object({
  list: builtinIconSetItemSchema.array(),
  total: z.number().int().min(0),
  page: z.number().int().min(1),
  pageSize: z.number().int().min(1),
})

export const iconSetRenderableIconSchema = z.object({
  icon: z.string().regex(iconifyIconNamePattern, '图标名称无效'),
  prefix: iconSetPrefixSchema,
  name: iconSetIconNameSchema,
  setName: z.string().min(1),
  body: z.string().min(1),
  width: z.number().int().min(1),
  height: z.number().int().min(1),
})

export const builtinIconListResponseSchema = z.object({
  list: iconSetRenderableIconSchema.array(),
  total: z.number().int().min(0),
  page: z.number().int().min(1),
  pageSize: z.number().int().min(1),
})

export const customIconSetSchema = z.object({
  prefix: iconSetPrefixSchema,
  name: z.string().min(1),
  description: z.string().nullable(),
  iconCount: z.number().int().min(0),
  createdAt: z.iso.datetime(),
  updatedAt: z.iso.datetime(),
})

export const customIconSetListResponseSchema = z.object({
  list: customIconSetSchema.array(),
  total: z.number().int().min(0),
  page: z.number().int().min(1),
  pageSize: z.number().int().min(1),
})

export const customIconItemSchema = iconSetRenderableIconSchema.extend({
  createdAt: z.iso.datetime(),
  updatedAt: z.iso.datetime(),
})

export const customIconListResponseSchema = z.object({
  list: customIconItemSchema.array(),
  total: z.number().int().min(0),
  page: z.number().int().min(1),
  pageSize: z.number().int().min(1),
})

const customIconSetDescriptionInputSchema = z
  .union([z.string().trim().max(customIconSetDescriptionMaxLength), z.null()])
  .optional()

export const customIconSetCreateSchema = z.object({
  prefix: iconSetPrefixSchema,
  name: z.string().trim().min(1, '请输入图标集名称').max(customIconSetNameMaxLength),
  description: customIconSetDescriptionInputSchema.transform((value) => value ?? null),
})

export const customIconSetUpdateSchema = z
  .object({
    name: z.string().trim().min(1, '请输入图标集名称').max(customIconSetNameMaxLength).optional(),
    description: customIconSetDescriptionInputSchema,
  })
  .refine((value) => value.name !== undefined || value.description !== undefined, {
    message: '至少修改一个字段',
  })

export const customIconDuplicateStrategySchema = z.enum(['skip', 'replace'], '重复处理策略无效')

export const iconSetRenameIconSchema = z.object({
  name: iconSetIconNameSchema,
})

export const customIconUploadSkippedSchema = z.object({
  name: iconSetIconNameSchema,
  sourceFilename: z.string().min(1),
  reason: z.literal('duplicate'),
})

export const customIconUploadFailedSchema = z.object({
  sourceFilename: z.string().min(1),
  message: z.string().min(1),
})

export const customIconUploadResponseSchema = z.object({
  created: customIconItemSchema.array(),
  replaced: customIconItemSchema.array(),
  skipped: customIconUploadSkippedSchema.array(),
  failed: customIconUploadFailedSchema.array(),
})

export type IconSetListQuery = z.infer<typeof iconSetListQuerySchema>
export type IconSetIconListQuery = z.infer<typeof iconSetIconListQuerySchema>
export type BuiltinIconSetItem = z.infer<typeof builtinIconSetItemSchema>
export type BuiltinIconSetListResponse = z.infer<typeof builtinIconSetListResponseSchema>
export type BuiltinIconItem = z.infer<typeof iconSetRenderableIconSchema>
export type BuiltinIconListResponse = z.infer<typeof builtinIconListResponseSchema>
export type CustomIconSet = z.infer<typeof customIconSetSchema>
export type CustomIconSetListResponse = z.infer<typeof customIconSetListResponseSchema>
export type CustomIconItem = z.infer<typeof customIconItemSchema>
export type CustomIconListResponse = z.infer<typeof customIconListResponseSchema>
export type CustomIconSetCreateInput = z.infer<typeof customIconSetCreateSchema>
export type CustomIconSetUpdateInput = z.infer<typeof customIconSetUpdateSchema>
export type CustomIconDuplicateStrategy = z.infer<typeof customIconDuplicateStrategySchema>
export type CustomIconUploadResponse = z.infer<typeof customIconUploadResponseSchema>
export type IconSetRenameIconInput = z.infer<typeof iconSetRenameIconSchema>

  'custom_icon_sets',

    id: uuidPrimaryKeyColumn(),
    prefix: text('prefix').notNull(),
    name: text('name').notNull(),
    description: text('description'),
    ...auditTimestamps(),

  'custom_icon_set_icons',
  {
    id: uuidPrimaryKeyColumn(),
    setId: uuid('set_id')
      .notNull()
      .references(() => customIconSets.id),
    name: text('name').notNull(),
    body: text('body').notNull(),
    width: integer('width').notNull(),
    height: integer('height').notNull(),
    palette: boolean('palette').notNull().default(false),
    ...auditTimestamps(),
  },
  (table) => [
    uniqueIndex('custom_icon_set_icons_set_name_active_unique')
      .on(table.setId, table.name)
      .where(sql`${table.deletedAt} IS NULL`),
    index('custom_icon_set_icons_set_id_idx').on(table.setId),

import { basename } from 'node:path/posix'
import { SVG, cleanupSVG, isEmptyColor, parseColors, runSVGO } from '@iconify/tools'
import { iconifyIconNamePartPatternSource } from '@rev30/contracts'
import { CustomSvgInvalidError } from './errors'

const svgExtensionPattern = /\.svg$/i
const separatorPattern = /[\s_]+/g
const invalidSeparatorPattern = /-+/g
const iconNamePattern = new RegExp(`^${iconifyIconNamePartPatternSource}$`)
const colorReferencePattern = /url\(/i

export type ParsedSvgIcon = {
  name: string
  body: string
  width: number
  height: number
  palette: boolean
}

function normalizeColor(value: string) {
  return value.trim().toLowerCase()
}

function isBlackColor(value: string) {
  return ['#000', '#000000', 'black', 'rgb(0,0,0)', 'rgb(0, 0, 0)'].includes(normalizeColor(value))
}

export function normalizeSvgIconName(filename: string) {
  const name = basename(filename)
    .replace(svgExtensionPattern, '')
    .trim()
    .toLowerCase()
    .replace(separatorPattern, '-')
    .replace(invalidSeparatorPattern, '-')
    .replace(/^-|-$/g, '')

  if (!iconNamePattern.test(name)) {
    throw new CustomSvgInvalidError('图标名称无效')
  }

  return name
}

export async function parseSvgIcon(filename: string, content: string): Promise<ParsedSvgIcon> {
  const name = normalizeSvgIconName(filename)

  try {
    const svg = new SVG(content)
    cleanupSVG(svg)

    let palette = false
    const colors = new Set<string>()

    parseColors(svg, {
      defaultColor: 'currentColor',
      callback: (_attr, colorString, color) => {
        if (!color || isEmptyColor(color)) {
          return colorString
        }

        if (colorReferencePattern.test(colorString)) {
          palette = true
          return colorString
        }

        const normalized = normalizeColor(colorString)
        colors.add(normalized)

        if (normalized === 'currentcolor' || isBlackColor(normalized)) {
          return 'currentColor'
        }

        palette = true
        return colorString
      },
    })

    if (colors.size > 1) {
      palette = true
    }

    runSVGO(svg)

    const exported = svg.getIcon()

    if (!exported.body) {
      throw new CustomSvgInvalidError()
    }

    return {
      name,
      body: exported.body,
      width: exported.width ?? 16,
      height: exported.height ?? 16,
      palette,
    }
  } catch (error) {
    if (error instanceof CustomSvgInvalidError) {
      throw error
    }

    throw new CustomSvgInvalidError()
  }
}

      const replaced: CustomIconItem[] = []

          const parsed = await parseSvgIcon(file.filename, file.content)
          const existing = await repository.findIcon(set.id, parsed.name)

          if (existing && input.duplicateStrategy === 'skip') {

              name: parsed.name,
              sourceFilename: file.filename,
              reason: 'duplicate',

            continue
          }

            ? await repository.replaceIcon(set, existing.id, parsed)
            : await repository.createIcon(set, parsed)

          if (existing) {
            replaced.push(saved)
          } else {
            created.push(saved)
          }
        } catch (error) {
          failed.push({
            sourceFilename: file.filename,
            message: error instanceof Error ? error.message : 'SVG 无效',
          })
        }
      }

      return { created, replaced, skipped, failed }
    },
    renameIcon: repository.renameIcon,
    deleteIcon: repository.softDeleteIcon,
    exportIconSet: async (prefix: string) => {
      const icons = await repository.listAllIconsForExport(prefix)
